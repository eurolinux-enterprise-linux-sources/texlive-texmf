\begin{eqnarray}
\arcsin x & = & -\arcsin(-x) = \frac{\pi}{2} - \arccos x = 
	       \left[\arccos\sqrt{1-x^2}\,\right]\nonumber\\
	  & = & \arctan\frac{x}{\sqrt{1-x^2}} =
	       \left[\mbox{\rm arccot}\frac{\sqrt{1-x^2}}{x}\right]
\end{eqnarray}
