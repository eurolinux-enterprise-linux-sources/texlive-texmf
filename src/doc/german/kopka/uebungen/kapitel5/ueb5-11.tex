\begin{equation}
\lim_{x\to0}\frac{\sqrt{1+x} - 1}{x} =
\lim_{x\to0}\frac{(\sqrt{1+x} - 1)(\sqrt{1+x} + 1)}{x(\sqrt{1+x} + 1)} =
\lim_{x\to0}\frac{1}{\sqrt{1+x} + 1} = \frac{1}{2}
\end{equation}
