  {\renewcommand{\PstGridThreeDHookNode}{%
     \PstGridThreeDNodeProcessor{blue}}%
   \PstGridThreeD[gridcolor=blue,GridThreeDZPos=4](1,8,1)}%
  {\renewcommand{\PstGridThreeDHookNode}{%
     \PstGridThreeDNodeProcessor{red}}%
   \PstGridThreeD[gridcolor=red,%
                  GridThreeDXPos=2,GridThreeDZPos=2](1,4,2)}
  {\renewcommand{\PstGridThreeDHookNode}{%
     \PstGridThreeDNodeProcessor{green}}%
   \PstGridThreeD[gridcolor=green,GridThreeDYPos=7](2,2,2)}}}
