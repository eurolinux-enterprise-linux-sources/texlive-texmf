  \pscustom[linecolor=red]{%
    \translate(#1,#2)
    \rotate{(#2,#1)}    % Vector in direction of field at x=#1,y=#2
  \psline(-.25,0)(.25,0)}}

      \ifnum\N=0
        \ifnum\n=0 \else\drawseg{\n}{\N}\fi
      \else
        \drawseg{\n}{\N}
      \fi}}
