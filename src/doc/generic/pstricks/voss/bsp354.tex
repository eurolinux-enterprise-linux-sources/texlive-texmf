\usepackage{pstricks,pst-plot,pst-math,pstricks-add}

\makeatletter
\def\psGaussFill{\pst@object{psGaussFill}}
\def\psGaussFill@i(#1,#2)(#3,#4)#5#6{{
  \pst@killglue
  \use@par
  \pscustom[linestyle=none]{%
    \psline(#4,0)(#3,0)
    \psplot[liftpen=1]{#3}{#4}{x #5 #6 GAUSS}%
  }%
  \psplot[fillstyle=none]{#1}{#2}{x #5 #6 GAUSS}%
}\ignorespaces}
\makeatother

\begin{document}


\psset{yunit=7.5cm, xunit=2cm}
\begin{pspicture}(-4,-0.1)(4,0.6)
  \psaxes[dy=0.2,Dy=0.2,linewidth=0.5pt]{->}(0,0)(-3.5,-0.1)(3.5,0.5)
  \psset{fillstyle=solid,plotpoints=100,linewidth=1pt}
  \psGaussFill[fillcolor=red](-3,3)(-2,-1.25){0}{0.9}
  \psGaussFill[fillcolor=blue](-3,3)(.5,1.5){0.5}{0.9}
  \psGaussFill[fillcolor=red,linestyle=none](-3,-1)(-2,-1.25){0}{0.9}
\end{pspicture}
