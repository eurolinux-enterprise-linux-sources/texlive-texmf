%%% A patch for mfpic 0.9 2006/05/26
%%%
\def\mfp@versionpatched{0.9}%
\def\mfp@patchinfo{%
  \Mfpic@warn{This patch is only for mfpic version \mfp@versionpatched!\@nl}}%
\let\mfp@dopatches\endinput
\ifx\mfpfileversion\mfp@versionpatched
  \def\mfp@patchinfo{}%
  \def\mfp@dopatches{}%
\fi
\mfp@patchinfo
\mfp@dopatches
\endinput
