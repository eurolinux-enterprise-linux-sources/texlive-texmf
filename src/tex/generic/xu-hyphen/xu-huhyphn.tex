% xu-huhyphn.tex
% Wrapper for XeTeX to read huhyphn.tex
% Jonathan Kew, 2006-08-17
% Public domain

\begingroup

\expandafter\ifx\csname XeTeXrevision\endcsname\relax
\else

  \let\PATTERNS=\patterns
  \def\patterns{% at the \patterns command in huhyphn.tex...
    \endgroup % end group containing huhyphn.tex's local definitions
    \begingroup % and start our own (to match \endgroup in huhyphn.tex)

    \input xu-t1.tex % map T1 codes to Unicode characters

    \XeTeXinputencoding "bytes" % read the rest of the file as raw bytes
    \PATTERNS % and then load the real patterns
  }

\fi

\input huhyphn.tex

\endgroup
\endinput
